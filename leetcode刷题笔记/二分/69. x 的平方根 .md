`while(r - l > eps)`

```c++
class Solution {
public:
    int mySqrt(int x) {
        double l = 0 , r = x;
        double eps = 1e-8;
        while(r - l > eps)
        {
            double mid = (l + r) / 2;
            if(mid * mid > x) r = mid;
            else l = mid;
        } 
        return (int)r;
    }
};
```
